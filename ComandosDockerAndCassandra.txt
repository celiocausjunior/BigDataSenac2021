CREATE KEYSPACE senac WITH replication = {'class':'SimpleStrategy', 'replication_factor' : 2};
describe keyspaces;
use senac;


CREATE TABLE covidCidades(
   id int PRIMARY KEY,
   regiao text,
   estado text,
   municipio text,
   coduf text,
   codmun text,
   data date,
   casosAcumulado varint,
   casosNovos varint,
   obitosAcumulado varint
   );
   
   



COPY senac.covidCidades (id,regiao,estado,municipio,coduf,codmun,data,casosAcumulado,casosNovos,obitosAcumulado) FROM 'covid19_formated_utf.csv' WITH DELIMITER=';' AND HEADER=TRUE;
