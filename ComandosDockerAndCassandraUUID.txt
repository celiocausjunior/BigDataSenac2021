CREATE KEYSPACE senac WITH replication = {'class':'SimpleStrategy', 'replication_factor' : 2};


describe keyspaces;

use senac;

CREATE TABLE covidCidades(
 regiao text,
   estado text,
   municipio text,
   coduf text,
   codmun text,
   data date,
   casosAcumulado varint,
   casosNovos varint,
   obitosAcumulado varint,
   id uuid PRIMARY KEY
   );

   
   



COPY senac.covidCidades (regiao,estado,municipio,coduf,codmun,data,casosAcumulado,casosNovos,obitosAcumulado,id) FROM 'export.csv' WITH DELIMITER=',' AND HEADER=TRUE;
